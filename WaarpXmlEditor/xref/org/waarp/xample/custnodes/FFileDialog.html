45  /**
46   * FFileDialog is a file dialog, which can be used as a custom field editor for
47   * the
48   * {@link com.fg.xmleditor.FXBasicView}. Since FFileDialog is a JDialog, it can
49   * be used as both custom field
50   * editor and a global editor.
51   *
52   * @author Felix Golubov
53   * @version 2.0
54   *     <p>
55   *     Add support for Directory and different options on File Dialog
56   */

58  /*
59   * Implementation notices: Well, it would much better to make buttons of the JFileChooser invisible and create
60   * our own "OK" and "Cancel" buttons. The problem is that CDE/Motif UI doesn't allow to make buttons
61   * invisible. Hence all the complexities with addBtnListener(Component c) recursive method. Besides, when UI
62   * changes, all the child components of the JFileChooser are discarded, so addBtnListener has to be called
63   * from the updateUI() method.
64   */
65  
66  public class FFileDialog extends JDialog
67      implements ICellControl, ActionListener {
68    /**
69     *
70     */
71    private static final long serialVersionUID = -8775488630211100329L;
72    JPanel panel = new JPanel();
73    final FFileChooser fileChooser = new FFileChooser();
74    int mode = JFileChooser.FILES_ONLY;
75    transient Object data;
76    transient List masks;
77  
78    public FFileDialog() {
79      getContentPane().setLayout(new BorderLayout());
80      fileChooser.setCurrentDirectory(new File("."));
81      fileChooser.addActionListener(this);
82      fileChooser.setApproveButtonText("OK");
83      getContentPane().add(fileChooser, BorderLayout.CENTER);
84    }
85  
86    @Override
87    protected void processWindowEvent(final WindowEvent e) {
88      if (e.getID() == WindowEvent.WINDOW_CLOSING) {
89        cancel();
90      }
91      super.processWindowEvent(e);
92    }
93  
94    void cancel() {
95      dispose();
96    }
97  
98    @Override
99    public void actionPerformed(final ActionEvent e) {
100     if (!(e.getSource() instanceof JButton)) {
101       return;
102     }
103     final JButton btn = (JButton) e.getSource();
104     if ("OK".equals(btn.getText())) {
105       final File file = fileChooser.getSelectedFile();
106       data = file == null? "" : file.getAbsolutePath();
107       cancel();
108     } else if ("Cancel".equals(btn.getText())) {
109       cancel();
110     } else if ("Annuler".equals(btn.getText())) {
111       cancel();
112     }
113   }
114 
115   @Override
116   public void initCellControl(final boolean isEditor) {
117     fileChooser.addBtnListener(fileChooser);
118   }
119 
120   @Override
121   public void updateCellControl(final boolean isEditor, final boolean enabled,
122                                 final boolean editable, final Object data,
123                                 final Params params) {
124     pack();
125     this.data = data;
126     if (data != null) {
127       final String path = data.toString().trim();
128       if (path.length() > 0) {
129         try {
130           final File file = new File(path);
131           fileChooser.setSelectedFile(file);
132         } catch (final Exception ex) {
133           fileChooser.setSelectedFile(null);
134         }
135       } else {
136         fileChooser.setSelectedFile(null);
137       }
138     } else {
139       fileChooser.setSelectedFile(null);
140     }
141     final List list = params.getList();
142     boolean sameMasks = true;
143     if (masks == null || list.size() != masks.size()) {
144       sameMasks = false;
145     } else {
146       for (int i = 0; i < list.size() && sameMasks; i++) {
147         if (!list.get(i).equals(masks.get(i))) {
148           sameMasks = false;
149           break;
150         }
151       }
152     }
153     if (sameMasks) {
154       return;
155     }
156     masks = list;
157     fileChooser.resetChoosableFileFilters();
158     for (int i = 0; i < masks.size(); i++) {
159       fileChooser
160           .addChoosableFileFilter(new FileExtFilter((String) masks.get(i)));
161     }
162   }
163 
164   @Override
165   public Object getData() {
166     return data;
167   }

169   class FileExtFilter extends FileFilter {
170     final HashSet<String> extensions = new HashSet<String>();
171     String description = "";
172 
173     FileExtFilter(final String mask) {
174       final StringTokenizer st = new StringTokenizer(mask, "|");
175       while (st.hasMoreElements()) {
176         String token = st.nextToken().trim();
177         if (st.hasMoreElements()) {
178           if (token.length() == 0) {
179             // no extension
180           } else {
181             if (token.startsWith("null")) {
182               if (token.length() > 4) {
183                 token = token.substring(5).trim();
184               } else {
185                 token = "";
186               }
187             }
188             if (token.length() > 0) {
189               extensions.add(token.toLowerCase());
190             }
191           }
192         } else {
193           final String type = token.substring(0, 1);
194           final int val = Integer.parseInt(type);
195           if (val == 0) {
196             mode = JFileChooser.FILES_ONLY;
197           } else if (val == 1) {
198             mode = JFileChooser.DIRECTORIES_ONLY;
199           } else {
200             mode = JFileChooser.FILES_AND_DIRECTORIES;
201           }
202           description = token.substring(1);
203         }
204       }
205       fileChooser.setFileSelectionMode(mode);
206     }
207 
208     @Override
209     public boolean accept(final File f) {
210       if (f.isDirectory()) {
211         return true;
212       }
213       if (f.isFile() && mode == JFileChooser.DIRECTORIES_ONLY) {
214         return false;
215       }
216       if (extensions.isEmpty()) {
217         return true;
218       }
219       final String ext;
220       final String name = f.getName();
221       final int i = name.lastIndexOf('.');
222       if (i > 0 && i < name.length() - 1) {
223         ext = name.substring(i + 1).toLowerCase();
224         return extensions.contains(ext);
225       } else {
226         return false;
227       }
228     }
229 
230     @Override
231     public String getDescription() {
232       return description;
233     }
234   }

236   class FFileChooser extends JFileChooser {
237     /**
238      *
239      */
240     private static final long serialVersionUID = -4161876854413783100L;
241 
242     public void addBtnListener(final Component c) {
243       if (c instanceof JButton) {
244         ((JButton) c).addActionListener(FFileDialog.this);
245       }
246       if (c instanceof Container) {
247         final Component[] children = ((Container) c).getComponents();
248         for (final Component element : children) {
249           addBtnListener(element);
250         }
251       }
252     }
253 
254     @Override
255     public void updateUI() {
256       setAcceptAllFileFilterUsed(getAcceptAllFileFilter() == null);
257       super.updateUI();
258       setAcceptAllFileFilterUsed(true);
259       addBtnListener(this);
260     }
261   }
