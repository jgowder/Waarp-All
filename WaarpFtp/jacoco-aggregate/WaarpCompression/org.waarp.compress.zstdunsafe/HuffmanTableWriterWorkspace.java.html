package org.waarp.compress.zstdunsafe;

class HuffmanTableWriterWorkspace {
  // for encoding weights
  public final byte[] weights = new byte[Huffman.MAX_SYMBOL];
  // the weight for the last symbol is implicit

  // for compressing weights
  public final int[] counts = new int[Huffman.MAX_TABLE_LOG + 1];
  public final short[] normalizedCounts = new short[Huffman.MAX_TABLE_LOG + 1];
  public final FseCompressionTable fseTable =
      new FseCompressionTable(Huffman.MAX_FSE_TABLE_LOG, Huffman.MAX_TABLE_LOG);
}
