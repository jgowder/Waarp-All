154   /**
155    * Set the new opened Channel (from channelConnected of {@link
156    * DataNetworkHandler})
157    *
158    * @param channel
159    * @param dataNetworkHandler
160    */
161   public void setOpenedDataChannel(final Channel channel,
162                                    final DataNetworkHandler dataNetworkHandler) {
163     logger.debug("SetOpenedDataChannel: " +
164                  (channel != null? channel.remoteAddress() : "no channel"));
165     if (channel != null) {
166       session.getDataConn().setDataNetworkHandler(dataNetworkHandler);
167       waitForOpenedDataChannel.setChannel(channel);
168       waitForOpenedDataChannel.setSuccess();
169     } else {
170       waitForOpenedDataChannel.cancel();
171     }
172   }

174   /**
175    * Wait that the new opened connection is ready (same method in {@link
176    * FtpDataAsyncConn} from openConnection)
177    *
178    * @return the new opened Channel
179    *
180    * @throws InterruptedException
181    */
182   public Channel waitForOpenedDataChannel() throws InterruptedException {
183     Channel channel = null;
184     if (waitForOpenedDataChannel.awaitOrInterruptible()) {
185       if (waitForOpenedDataChannel.isSuccess()) {
186         channel = waitForOpenedDataChannel.channel();
187       } else {
188         logger.warn("data connection is in error");
189       }
190     } else {
191       logger.warn("Timeout occurs during data connection");
192       throw new InterruptedException("Cannot get data connection");
193     }
194     return channel;
195   }

202       waitForOpenedDataChannel.cancel();

204     waitForOpenedDataChannel = new WaarpChannelFuture(true);

207   /**
208    * Wait for the client to be connected (Passive) or Wait for the server to
209    * be
210    * connected to the client (Active)
211    *
212    * @return True if the connection is OK
213    *
214    * @throws Reply425Exception
215    */
216   public synchronized boolean openDataConnection() throws Reply425Exception {
217     // Prepare this Data channel to be closed ;-)
218     // In fact, prepare the future close op which should occur since it is
219     // now opened
220     if (commandSetup != null) {
221       commandSetup.cancel();
222     }
223     commandSetup = new WaarpFuture(true);
224     final FtpDataAsyncConn dataAsyncConn = session.getDataConn();
225     if (!dataAsyncConn.isStreamFile()) {
226       // FIXME isActive or isDNHReady ?
227       if (dataAsyncConn.isActive()) {
228         // Already connected
229         logger.debug("Connection already open");
230         session.setReplyCode(ReplyCode.REPLY_125_DATA_CONNECTION_ALREADY_OPEN,
231                              dataAsyncConn.getType().name() +
232                              " mode data connection already open");
233         return true;
234       }
235     } else {
236       // Stream, Data Connection should not be opened
237       if (dataAsyncConn.isActive()) {
238         logger.error(
239             "Connection already open but should not since in Stream mode");
240         setTransferAbortedFromInternal(false);
241         throw new Reply425Exception(
242             "Connection already open but should not since in Stream mode");
243       }
244     }
245     // Need to open connection
246     session.setReplyCode(ReplyCode.REPLY_150_FILE_STATUS_OKAY,
247                          "Opening " + dataAsyncConn.getType().name() +
248                          " mode data connection");
249     if (dataAsyncConn.isPassiveMode()) {
250       if (!dataAsyncConn.isBind()) {
251         // No passive connection prepared
252         throw new Reply425Exception("No passive data connection prepared");
253       }
254       // Wait for the connection to be done by the client
255       logger.debug("Passive mode standby");
256       try {
257         dataChannel = waitForOpenedDataChannel();
258         dataAsyncConn.setNewOpenedDataChannel(dataChannel);
259       } catch (final InterruptedException e) {//NOSONAR
260         logger.warn("Connection abort in passive mode", e);
261         // Cannot open connection
262         throw new Reply425Exception("Cannot open passive data connection");
263       }
264       logger.debug("Passive mode connected");
265     } else {
266       // Wait for the server to be connected to the client
267       final InetAddress inetAddress =
268           dataAsyncConn.getLocalAddress().getAddress();
269       final InetSocketAddress inetSocketAddress =
270           dataAsyncConn.getRemoteAddress();
271       if (session.getConfiguration().getFtpInternalConfiguration()
272                  .hasFtpSession(inetAddress, inetSocketAddress)) {
273         throw new Reply425Exception(
274             "Cannot open active data connection since remote address is already in use: " +
275             inetSocketAddress);
276       }
277       logger.debug("Active mode standby");
278       final Bootstrap bootstrap =
279           session.getConfiguration().getFtpInternalConfiguration()
280                  .getActiveBootstrap(session.isDataSsl());
281       session.getConfiguration()
282              .setNewFtpSession(inetAddress, inetSocketAddress, session);
283       // Set the session for the future dataChannel
284       final String mylog = session.toString();
285       logger.debug(
286           "DataConn for: " + session.getCurrentCommand().getCommand() + " to " +
287           inetSocketAddress);
288       final ChannelFuture future =
289           bootstrap.connect(inetSocketAddress, dataAsyncConn.getLocalAddress());
290       try {
291         future.await(session.getConfiguration().getTimeoutCon());
292       } catch (final InterruptedException e1) {//NOSONAR
293         SysErrLogger.FAKE_LOGGER.ignoreLog(e1);
294       }
295       if (!future.isSuccess()) {
296         logger.warn(
297             "Connection abort in active mode from future while session: " +
298             session + "\nTrying connect to: " + inetSocketAddress + " From: " +
299             dataAsyncConn.getLocalAddress() + "\nWas: " + mylog,
300             future.cause());
301         // Cannot open connection
302         session.getConfiguration()
303                .delFtpSession(inetAddress, inetSocketAddress);
304         throw new Reply425Exception("Cannot open active data connection");
305       }
306       try {
307         dataChannel = waitForOpenedDataChannel();
308         dataAsyncConn.setNewOpenedDataChannel(dataChannel);
309       } catch (final InterruptedException e) {//NOSONAR
310         logger.warn("Connection abort in active mode", e);
311         // Cannot open connection
312         session.getConfiguration()
313                .delFtpSession(inetAddress, inetSocketAddress);
314         throw new Reply425Exception("Cannot open active data connection");
315       }
316       logger.debug("Active mode connected");
317     }
318     if (dataChannel == null) {
319       // Cannot have a new Data connection since shutdown
320       if (!dataAsyncConn.isPassiveMode()) {
321         session.getConfiguration().getFtpInternalConfiguration()
322                .delFtpSession(dataAsyncConn.getLocalAddress().getAddress(),
323                               dataAsyncConn.getRemoteAddress());
324       }
325       throw new Reply425Exception("Cannot open data connection, shuting down");
326     }
327     return true;
328   }

332   /**
333    * Run the command from an executor
334    */
335   private void runExecutor() {
336     endOfCommand = new WaarpFuture(true);
337     try {
338       session.getDataConn().getDataNetworkHandler()
339              .setFtpTransfer(executingCommand);
340     } catch (final FtpNoConnectionException ignored) {
341       // nothing
342     }
343 
344     if (!waitForOpenedDataChannel.awaitOrInterruptible()) {
345       commandFinishing.cancel();
346       endOfCommand.cancel();
347       return;
348     }
349     waitForOpenedDataChannel.channel().config().setAutoRead(true);
350     // Run the command
351     if (executorService == null) {
352       executorService = Executors.newSingleThreadExecutor();
353     }
354     executorService.execute(new FtpTransferExecutor(session, executingCommand));
355     commandFinishing.awaitOrInterruptible();
356     if (commandFinishing.isFailed()) {
357       endOfCommand.cancel();
358     }
359   }

361   /**
362    * Add a new transfer to be executed. This is to be called from Command
363    * after
364    * connection is opened and before
365    * answering to the client that command is ready to be executed (for Store
366    * or
367    * Retrieve like operations).
368    *
369    * @param command
370    * @param file
371    */
372   public void setNewFtpTransfer(final FtpCommandCode command,
373                                 final FtpFile file) {
374     isExecutingCommandFinished = false;
375     commandFinishing = new WaarpFuture(true);
376     logger.debug("setNewCommand: {}", command);
377     setDataNetworkHandlerReady();
378     executingCommand = new FtpTransfer(command, file);
379     runExecutor();
380     commandFinishing = null;
381     commandSetup.setSuccess();
382     if (!session.getDataConn().isStreamFile()) {
383       waitForOpenedDataChannel.channel().config().setAutoRead(false);
384     }
385   }

388    * Add a new transfer to be executed. This is to be called from Command
389    * after
390    * connection is opened and before
391    * answering to the client that command is ready to be executed (for List
392    * like
393    * operations).
394    *
395    * @param command
396    * @param list
397    * @param path as Original Path
398    */
399   public void setNewFtpTransfer(final FtpCommandCode command,
400                                 final List<String> list, final String path) {
401     isExecutingCommandFinished = false;
402     commandFinishing = new WaarpFuture(true);
403     logger.debug("setNewCommand: {}", command);
404     setDataNetworkHandlerReady();
405     executingCommand = new FtpTransfer(command, list, path);
406     runExecutor();
407     commandFinishing = null;
408     commandSetup.setSuccess();
409     if (!session.getDataConn().isStreamFile()) {
410       waitForOpenedDataChannel.channel().config().setAutoRead(false);
411     }
412     try {
413       session.getDataConn().getDataNetworkHandler().setFtpTransfer(null);
414     } catch (final FtpNoConnectionException ignored) {
415       // nothing
416     }
417   }

419   public boolean waitFtpTransferExecuting() {
420     boolean notFinished = true;
421     for (int i = 0; i < FtpInternalConfiguration.RETRYNB * 100; i++) {
422       if (isExecutingCommandFinished || commandFinishing == null ||
423           session.isCurrentCommandFinished() || commandFinishing != null &&
424                                                 commandFinishing
425                                                     .awaitOrInterruptible(
426                                                         FtpInternalConfiguration.RETRYINMS)) {
427         notFinished = false;
428         break;
429       }
430     }
431     return notFinished;
432   }

458   /**
459    * @return True if the current FtpTransfer is a Retrieve like transfer
460    *
461    * @throws FtpNoTransferException
462    * @throws CommandAbstractException
463    * @throws FtpNoFileException
464    */
465   boolean isExecutingRetrLikeTransfer()
466       throws FtpNoTransferException, CommandAbstractException,
467              FtpNoFileException {
468     return !session.isCurrentCommandFinished() && FtpCommandCode
469         .isRetrLikeCommand(getExecutingFtpTransfer().getCommand()) &&
470            getExecutingFtpTransfer().getFtpFile().isInReading();
471   }

473   /**
474    * Called when a transfer is finished from setEndOfTransfer
475    *
476    * @return True if it was already called before
477    *
478    * @throws FtpNoTransferException
479    */
480   private boolean checkFtpTransferStatus() throws FtpNoTransferException {
481     if (isCheckAlreadyCalled) {
482       logger.warn("Check: ALREADY CALLED");
483       return true;
484     }
485     if (isExecutingCommandFinished) {
486       // already done
487       logger.warn("Check: already Finished");
488       if (commandFinishing != null) {
489         commandFinishing.cancel();
490       }
491       throw new FtpNoTransferException("No transfer running");
492     }
493     if (!isDataNetworkHandlerReady) {
494       // already done
495       logger.warn("Check: already DNH not ready");
496       throw new FtpNoTransferException("No connection");
497     }
498     isCheckAlreadyCalled = true;
499     final FtpTransfer executedTransfer = getExecutingFtpTransfer();
500     logger.debug("Check: command {}", executedTransfer.getCommand());
501     // DNH is ready and Transfer is running
502     if (FtpCommandCode.isListLikeCommand(executedTransfer.getCommand())) {
503       if (executedTransfer.getStatus()) {
504         // Special status for List Like command
505         logger.debug("Check: List OK");
506         closeTransfer();
507         return false;
508       }
509       logger.debug("Check: List Ko");
510       abortTransfer();
511       return false;
512     } else if (FtpCommandCode
513         .isRetrLikeCommand(executedTransfer.getCommand())) {
514       final FtpFile file;
515       try {
516         file = executedTransfer.getFtpFile();
517       } catch (final FtpNoFileException e) {
518         logger.debug("Check: Retr no FtpFile for Retr");
519         abortTransfer();
520         return false;
521       }
522       try {
523         if (file.isInReading()) {
524           logger.debug("Check: Retr FtpFile still in reading KO");
525           abortTransfer();
526         } else {
527           logger.debug("Check: Retr FtpFile no more in reading OK");
528           closeTransfer();
529         }
530       } catch (final CommandAbstractException e) {
531         logger.warn("Retr Test is in Reading problem", e);
532         closeTransfer();
533       }
534       return false;
535     } else if (FtpCommandCode
536         .isStoreLikeCommand(executedTransfer.getCommand())) {
537       // logger.debug("Check: Store OK")
538       closeTransfer();
539       return false;
540     } else {
541       logger.warn("Check: Unknown command");
542       abortTransfer();
543     }
544     return false;
545   }

547   /**
548    * Abort the current transfer
549    */
550   private void abortTransfer() {
551     logger
552         .debug("Will abort transfer and write: ", new Exception("trace only"));
553     final FtpFile file;
554     FtpTransfer current = null;
555     try {
556       current = getExecutingFtpTransfer();
557       file = current.getFtpFile();
558       file.abortFile();
559     } catch (final FtpNoTransferException e) {
560       logger.warn("Abort problem", e);
561     } catch (final FtpNoFileException ignored) {
562       // nothing
563     } catch (final CommandAbstractException e) {
564       logger.warn("Abort problem", e);
565     }
566     if (current != null) {
567       current.setStatus(false);
568     }
569     endDataConnection();
570     session.setReplyCode(ReplyCode.REPLY_426_CONNECTION_CLOSED_TRANSFER_ABORTED,
571                          "Transfer aborted for " +
572                          (current == null? "Unknown command" :
573                              current.toString()));
574     if (current != null &&
575         !FtpCommandCode.isListLikeCommand(current.getCommand())) {
576       try {
577         session.getBusinessHandler().afterTransferDoneBeforeAnswer(current);
578       } catch (final CommandAbstractException e) {
579         session.setReplyCode(e);
580       }
581     }
582     finalizeExecution();
583   }

585   /**
586    * Finish correctly a transfer
587    */
588   private void closeTransfer() {
589     logger.debug("Will close transfer");
590     final FtpFile file;
591     FtpTransfer current = null;
592     try {
593       current = getExecutingFtpTransfer();
594       file = current.getFtpFile();
595       file.closeFile();
596     } catch (final FtpNoTransferException e) {
597       logger.warn("Close problem", e);
598     } catch (final FtpNoFileException ignored) {
599       // nothing
600     } catch (final CommandAbstractException e) {
601       logger.warn("Close problem", e);
602     }
603     if (current != null) {
604       current.setStatus(true);
605     }
606     if (session.getDataConn().isStreamFile()) {
607       endDataConnection();
608     }
609     session.setReplyCode(ReplyCode.REPLY_226_CLOSING_DATA_CONNECTION,
610                          "Transfer complete for " +
611                          (current == null? "Unknown command" :
612                              current.toString()));
613     if (current != null) {
614       if (!FtpCommandCode.isListLikeCommand(current.getCommand())) {
615         try {
616           session.getBusinessHandler().afterTransferDoneBeforeAnswer(current);
617         } catch (final CommandAbstractException e) {
618           session.setReplyCode(e);
619         }
620       } else {
621         // Special wait to prevent fast LIST following by STOR or RETR command
622         try {
623           Thread.sleep(FtpInternalConfiguration.RETRYINMS);
624         } catch (final InterruptedException e) {//NOSONAR
625           SysErrLogger.FAKE_LOGGER.ignoreLog(e);
626         }
627       }
628     }
629     finalizeExecution();
630   }

644   /**
645    * To enable abort from internal error
646    *
647    * @param write True means the message is write back to the control
648    *     command, false it is only prepared
649    */
650   public void setTransferAbortedFromInternal(final boolean write) {
651     logger.debug("Set transfer aborted internal {}", write);
652     abortTransfer();
653     if (write) {
654       session.getNetworkHandler().writeIntermediateAnswer();
655     }
656     if (endOfCommand != null) {
657       endOfCommand.cancel();
658     }
659   }

690   /**
691    * Finalize execution
692    */
693   private void finalizeExecution() {
694     // logger.debug("Finalize execution")
695     if (commandFinishing != null) {
696       commandFinishing.setSuccess();
697     }
698     isExecutingCommandFinished = true;
699     executingCommand = null;
700     resetWaitForOpenedDataChannel();
701   }
702 
703   /**
704    * End the data connection if any
705    */
706   private synchronized void endDataConnection() {
707     logger.debug("End Data connection");
708     if (isDataNetworkHandlerReady && dataChannel != null) {
709       WaarpNettyUtil
710           .awaitOrInterrupted(WaarpSslUtility.closingSslChannel(dataChannel));
711       isDataNetworkHandlerReady = false;
712       // logger.debug("waitForClosedDataChannel over")
713       dataChannel = null;
714     }
715   }

717   /**
718    * Clear the FtpTransferControl (called when the data connection must be
719    * over
720    * like from clear of
721    * {@link FtpDataAsyncConn}, abort from {@link ABOR} or ending control
722    * connection from {@link NetworkHandler}.
723    */
724   public void clear() {
725     // logger.debug("Clear Ftp Transfer Control")
726     endDataConnection();
727     finalizeExecution();
728     if (endOfCommand != null) {
729       endOfCommand.cancel();
730     }
731     if (waitForOpenedDataChannel != null) {
732       waitForOpenedDataChannel.cancel();
733     }
734     if (commandSetup != null) {
735       commandSetup.cancel();
736     }
737     if (executorService != null) {
738       executorService.shutdownNow();
739       executorService = null;
740     }
741   }
