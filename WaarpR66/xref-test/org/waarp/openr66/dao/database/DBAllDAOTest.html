198   public String getServerConfigFile() {
199     if (configFile != null) {
200       return configFile.getAbsolutePath();
201     }
202     ClassLoader classLoader = DBAllDAOTest.class.getClassLoader();
203     File file = new File(
204         classLoader.getResource("Linux/config/config-XXDb.xml").getFile());
205     if (!file.exists()) {
206       SysErrLogger.FAKE_LOGGER
207           .syserr("Cannot find in  " + file.getAbsolutePath());
208       fail("Cannot find " + file.getAbsolutePath());
209     }
210     String content = WaarpStringUtils.readFile(file.getAbsolutePath());
211     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getJdbcUrl());
212     String driver = getJDC().getDriverClassName();
213     String target = "notfound";
214     String jdbcUrl = getJDC().getJdbcUrl();
215     if (driver.equalsIgnoreCase("org.mariadb.jdbc.Driver")) {
216       target = "mariadb";
217     } else if (driver.equalsIgnoreCase("org.h2.Driver")) {
218       target = "h2";
219     } else if (driver.equalsIgnoreCase("oracle.jdbc.OracleDriver")) {
220       target = "oracle";
221       jdbcUrl = "jdbc:oracle:thin:@//localhost:1521/test";
222       SysErrLogger.FAKE_LOGGER
223           .syserr(jdbcUrl + " while should be something like " + jdbcUrl);
224       throw new UnsupportedOperationException(
225           "Unsupported Test for Oracle since wrong JDBC driver");
226     } else if (driver.equalsIgnoreCase("org.postgresql.Driver")) {
227       target = "postgresql";
228     } else if (driver.equalsIgnoreCase("com.mysql.jdbc.Driver")) {
229       target = "mysql";
230     } else {
231       SysErrLogger.FAKE_LOGGER.syserr("Cannot find driver for " + driver);
232     }
233     content = content.replace("XXXJDBCXXX", jdbcUrl);
234     content = content.replace("XXXDRIVERXXX", target);
235     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getDriverClassName());
236     SysErrLogger.FAKE_LOGGER.sysout(target);
237     File fileTo = new File(TMP_R_66_CONF_CONFIG_XML);
238     fileTo.getParentFile().mkdirs();
239     FileWriter writer = null;
240     try {
241       writer = new FileWriter(fileTo);
242       writer.write(content);
243       writer.flush();
244     } catch (IOException e) {
245       e.printStackTrace();
246     } finally {
247       if (writer != null) {
248         FileUtils.close(writer);
249       }
250     }
251     configFile = fileTo;
252     File copy = new File(TMP_CONFIG_XML);
253     try {
254       FileUtils.copy(configFile, copy, false, false);
255     } catch (Reply550Exception e) {
256       e.printStackTrace();
257     }
258     return TMP_R_66_CONF_CONFIG_XML;
259   }

261   @Test
262   public void startStopServerR66() throws Exception {
263     String fileconf = null;
264     try {
265       fileconf = getServerConfigFile();
266     } catch (UnsupportedOperationException e) {
267       SysErrLogger.FAKE_LOGGER
268           .syserr("Database not supported by this test Start Stop R66", e);
269       Assume.assumeNoException(e);
270       return;
271     }
272     setUpBeforeClassMinimal(fileconf);
273     File copy = new File(TMP_CONFIG_XML);
274     if (copy.exists() && configFile != null && !configFile.exists()) {
275       try {
276         FileUtils.copy(copy, configFile, false, false);
277       } catch (Reply550Exception e) {
278         //e.printStackTrace();
279       }
280     }
281     String fileserver = fileconf;
282     if (!fileconf.startsWith("/")) {
283       int pos = fileconf.lastIndexOf('/');
284       if (pos >= 0) {
285         fileserver = fileconf.substring(pos + 1);
286       }
287     }
288     setUpBeforeClassServer(fileconf, fileserver, true);
289     setUpBeforeClassClient(fileserver);
290     Configuration.configuration.setTimeoutCon(100);
291     final DbHostAuth host = new DbHostAuth("hostas");
292     final SocketAddress socketServerAddress;
293     try {
294       socketServerAddress = host.getSocketAddress();
295     } catch (final IllegalArgumentException e) {
296       logger.error("Needs a correct configuration file as first argument");
297       return;
298     }
299     final byte scode = -1;
300 
301     // Shutdown server
302     logger.warn("Shutdown Server");
303     Configuration.configuration.setTimeoutCon(100);
304     final R66Future future = new R66Future(true);
305     final ShutdownOrBlockJsonPacket node8 = new ShutdownOrBlockJsonPacket();
306     node8.setRestartOrBlock(false);
307     node8.setShutdownOrBlock(true);
308     node8.setKey(FilesystemBasedDigest.passwdCrypt(
309         Configuration.configuration.getServerAdminKey()));
310     final AbstractLocalPacket valid =
311         new JsonCommandPacket(node8, LocalPacketFactory.BLOCKREQUESTPACKET);
312     sendInformation(valid, socketServerAddress, future, scode, false,
313                     R66FiniteDualStates.SHUTDOWN, true);
314     Thread.sleep(200);
315 
316     tearDownAfterClassClient();
317     tearDownAfterClassMinimal();
318     ChannelUtils.exit();
319   }
