122   public String getServerConfigFile() {
123     if (configFile != null) {
124       logger.warn("ConfigFile already set to {]", configFile);
125       return configFile.getAbsolutePath();
126     }
127     logger.warn("Build configFile");
128     ClassLoader classLoader = ScenarioBaseBigFile.class.getClassLoader();
129     File file =
130         new File(classLoader.getResource(RESOURCES_SERVER_1_XML).getFile());
131     if (!file.exists()) {
132       SysErrLogger.FAKE_LOGGER
133           .syserr("Cannot find in  " + file.getAbsolutePath());
134       fail("Cannot find " + file.getAbsolutePath());
135     }
136     String content = WaarpStringUtils.readFile(file.getAbsolutePath());
137     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getJdbcUrl());
138     String driver = getJDC().getDriverClassName();
139     String target = "notfound";
140     String jdbcUrl = getJDC().getJdbcUrl();
141     if (driver.equalsIgnoreCase("org.mariadb.jdbc.Driver")) {
142       target = "mariadb";
143     } else if (driver.equalsIgnoreCase("org.h2.Driver")) {
144       target = "h2";
145     } else if (driver.equalsIgnoreCase("oracle.jdbc.OracleDriver")) {
146       target = "oracle";
147       jdbcUrl = "jdbc:oracle:thin:@//localhost:1521/test";
148       SysErrLogger.FAKE_LOGGER
149           .syserr(jdbcUrl + " while should be something like " + jdbcUrl);
150       throw new UnsupportedOperationException(
151           "Unsupported Test for Oracle since wrong JDBC driver");
152     } else if (driver.equalsIgnoreCase("org.postgresql.Driver")) {
153       target = "postgresql";
154     } else if (driver.equalsIgnoreCase("com.mysql.jdbc.Driver")) {
155       target = "mysql";
156     } else {
157       SysErrLogger.FAKE_LOGGER.syserr("Cannot find driver for " + driver);
158     }
159     content = content.replace("XXXJDBCXXX", jdbcUrl);
160     content = content.replace("XXXDRIVERXXX", target);
161     SysErrLogger.FAKE_LOGGER.sysout(getJDC().getDriverClassName());
162     SysErrLogger.FAKE_LOGGER.sysout(target);
163     File fileTo = new File(TMP_R66_CONFIG_R1);
164     fileTo.getParentFile().mkdirs();
165     FileWriter writer = null;
166     try {
167       writer = new FileWriter(fileTo);
168       writer.write(content);
169       writer.flush();
170     } catch (IOException e) {
171       e.printStackTrace();
172     } finally {
173       if (writer != null) {
174         FileUtils.close(writer);
175       }
176     }
177     configFile = fileTo;
178     logger.warn("Config file created at {}", fileTo);
179     File copy = new File(TMP_CONFIG_XML);
180     try {
181       FileUtils.copy(configFile, copy, false, false);
182       logger.warn("Copy from {} to {}", configFile, copy);
183     } catch (Reply550Exception e) {
184       e.printStackTrace();
185     }
186     logger.warn("Copy from {} to {} and return {}", configFile, copy,
187                 TMP_R66_CONFIG_R1);
188     return TMP_R66_CONFIG_R1;
189   }

233     }
234     initiateDb(configFile.getAbsolutePath());
235     initiateDb(SERVER_2_XML);
236 
237     Processes.finalizeProject(project);

266   public static int startServer(String serverConfig) throws Exception {
267     final File file2;
268     if (serverConfig.charAt(0) == '/') {
269       file2 = new File(serverConfig);
270     } else {
271       file2 = new File(dirResources, serverConfig);
272     }
273     if (file2.exists()) {
274       System.err.println("Find server file: " + file2.getAbsolutePath());
275       final String[] argsServer = {
276           file2.getAbsolutePath()
277       };
278       // global ant project settings
279       project = Processes.getProject(homeDir);
280       Processes.executeJvm(project, homeDir, R66Server.class, argsServer, true);
281       int pid = Processes
282           .getPidOfRunnerCommandLinux("java", R66Server.class.getName(), PIDS);
283       PIDS.add(pid);
284       logger.warn("Start Done: {}", pid);
285       return pid;
286     } else {
287       System.err.println("Cannot find server file: " + file2.getAbsolutePath());
288       fail("Cannot find server file");
289       return 999999;
290     }

328   private void checkMemory() {
329     Runtime runtime = Runtime.getRuntime();
330     long newUsedMemory = runtime.totalMemory() - runtime.freeMemory();
331     if (newUsedMemory > MAX_USED_MEMORY) {
332       logger.warn("Used Memory > 512MB {} {}", usedMemory / 1048576.0,
333                   newUsedMemory / 1048576.0);

337   private void testBigTransfer(boolean limit, String serverName, boolean direct,
338                                boolean recv)
339       throws IOException, InterruptedException {
340     if (!SERVER1_IN_JUNIT && !direct) {
341       logger.warn("Only Direct is enabled in Client mode");
342       return;
343     }
344     Assume.assumeNotNull(networkTransaction);
345     if (limit) {
346       Configuration.configuration
347           .changeNetworkLimit(BANDWIDTH * 2, BANDWIDTH * 2, BANDWIDTH,
348                               BANDWIDTH, 1000);
349     } else {
350       Configuration.configuration.changeNetworkLimit(0, 0, 0, 0, 100000);
351     }
352     File baseDir;
353     if (recv && !"server1".equals(serverName)) {
354       baseDir = new File("/tmp/R66/scenario_big_file_limitbdw/R2/out/");
355     } else {
356       baseDir = new File("/tmp/R66/scenario_big_file_limitbdw/R1/out/");
357     }
358     File fileOut = new File(baseDir, "hello");
359     final File outHello =
360         generateOutFile(fileOut.getAbsolutePath(), LARGE_SIZE);
361     long timestart = System.currentTimeMillis();
362     final R66Future future = new R66Future(true);
363     if (direct) {
364       if (recv) {
365         final TestRecvThroughHandler handler = new TestRecvThroughHandler();
366         final TestRecvThroughClient transaction =
367             new TestRecvThroughClient(future, handler, serverName, "hello",
368                                       "recvthrough", "Test Big RecvThrough",
369                                       true, BLOCK_SIZE, networkTransaction);
370         transaction.setNormalInfoAsWarn(false);
371         transaction.run();
372       } else {
373         final TestTransferNoDb transaction =
374             new TestTransferNoDb(future, serverName, "hello", "sendany",
375                                  "Test Big Send", true, BLOCK_SIZE,
376                                  DbConstantR66.ILLEGALVALUE,
377                                  networkTransaction);
378         transaction.setNormalInfoAsWarn(false);
379         transaction.run();
380       }
381       future.awaitOrInterruptible();
382       assertTrue(future.isSuccess());
383     } else {
384       String ruleName = recv? "recv" : "sendany";
385       final SubmitTransfer transaction =
386           new SubmitTransfer(future, serverName, "hello", ruleName,
387                              "Test Big " + ruleName, true, BLOCK_SIZE,
388                              DbConstantR66.ILLEGALVALUE, null);
389       transaction.run();
390       future.awaitOrInterruptible();
391       assertTrue(future.isSuccess());
392       final DbTaskRunner runner = future.getResult().getRunner();
393       if (runner != null) {
394         logger.info("Runner: {}", runner.toString());
395         waitForAllDone(runner);
396         dbTaskRunners.add(runner);
397       }
398     }
399     long timestop = System.currentTimeMillis();
400     logger.warn("Direct {}, Recv {}, LimitBandwidth {} " +
401                 "({} seconds,  {} MBPS vs {} " +
402                 "and {}) of size {} with block size {}", direct, recv, limit,
403                 (timestop - timestart) / 1000,
404                 LARGE_SIZE / 1000.0 / (timestop - timestart),
405                 Configuration.configuration.getServerGlobalReadLimit() /
406                 1000000.0,
407                 Configuration.configuration.getServerChannelReadLimit() /
408                 1000000.0, LARGE_SIZE, BLOCK_SIZE);
409     outHello.delete();
410     FileUtils.forceDeleteRecursiveDir(baseDir);
411     checkMemory();
412   }

582   private void waitForAllDone(DbTaskRunner runner) {
583     while (true) {
584       try {
585         DbTaskRunner checkedRunner = DbTaskRunner.reloadFromDatabase(runner);
586         if (checkedRunner.isAllDone()) {
587           logger.warn("DbTaskRunner done");
588           return;
589         } else if (checkedRunner.isInError()) {
590           logger.error("DbTaskRunner in error");
591           fail("DbTaskRunner in error");
592           return;
593         }
594         Thread.sleep(100);
595       } catch (InterruptedException e) {//NOSONAR
596         logger.error("Interrupted", e);
597         return;
598       } catch (WaarpDatabaseException e) {
599         logger.error("Cannot found DbTaskRunner", e);
600         return;
601       }
602     }
603   }
