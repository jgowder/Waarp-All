27  /**
28   * This enum class references all available tasks.
29   * <p>
30   * If a new task type is to be created, this is the place where it should be
31   * referenced.
32   */
33  public enum TaskType {
34    LOG, MOVE, MOVERENAME, COPY, COPYRENAME, EXEC, EXECMOVE, LINKRENAME, TRANSFER,
35    VALIDFILEPATH, DELETE, TAR, ZIP, EXECOUTPUT, RESCHEDULE, EXECJAVA, TRANSCODE,
36    SNMP, FTP, RENAME, RESTART, UNZEROED, CHMOD, CHKFILE, ICAP;
37  
38    private static final String UNVALID_TASK = "Unvalid Task: ";
39  
40    private static final String NAME_UNKNOWN = "name unknown: ";
41  
42    final int type;
43  
44    final String name;
45    /**
46     * Internal Logger
47     */
48    private static final WaarpLogger logger =
49        WaarpLoggerFactory.getLogger(TaskType.class);
50  
51    TaskType() {
52      type = ordinal();
53      name = name();
54    }
55  
56    /**
57     * @param type
58     * @param argRule
59     * @param delay
60     * @param session
61     *
62     * @return the corresponding AbstractTask
63     *
64     * @throws OpenR66RunnerErrorException
65     */
66    public static AbstractTask getTaskFromId(final TaskType type,
67                                             final String argRule,
68                                             final int delay,
69                                             final R66Session session)
70        throws OpenR66RunnerErrorException {
71      switch (type) {
72        case LOG:
73          return new LogTask(argRule, delay,
74                             session.getRunner().getFileInformation(), session);
75        case MOVE:
76          return new MoveTask(argRule, delay,
77                              session.getRunner().getFileInformation(), session);
78        case MOVERENAME:
79          return new MoveRenameTask(argRule, delay,
80                                    session.getRunner().getFileInformation(),
81                                    session);
82        case COPY:
83          return new CopyTask(argRule, delay,
84                              session.getRunner().getFileInformation(), session);
85        case COPYRENAME:
86          return new CopyRenameTask(argRule, delay,
87                                    session.getRunner().getFileInformation(),
88                                    session);
89        case EXEC:
90          return new ExecTask(argRule, delay,
91                              session.getRunner().getFileInformation(), session);
92        case EXECMOVE:
93          return new ExecMoveTask(argRule, delay,
94                                  session.getRunner().getFileInformation(),
95                                  session);
96        case LINKRENAME:
97          return new LinkRenameTask(argRule, delay,
98                                    session.getRunner().getFileInformation(),
99                                    session);
100       case TRANSFER:
101         return new TransferTask(argRule, delay,
102                                 session.getRunner().getFileInformation(),
103                                 session);
104       case VALIDFILEPATH:
105         return new ValidFilePathTask(argRule, delay,
106                                      session.getRunner().getFileInformation(),
107                                      session);
108       case DELETE:
109         return new DeleteTask(argRule, delay,
110                               session.getRunner().getFileInformation(),
111                               session);
112       case TAR:
113         return new TarTask(argRule, delay,
114                            session.getRunner().getFileInformation(), session);
115       case ZIP:
116         return new ZipTask(argRule, delay,
117                            session.getRunner().getFileInformation(), session);
118       case EXECOUTPUT:
119         return new ExecOutputTask(argRule, delay,
120                                   session.getRunner().getFileInformation(),
121                                   session);
122       case RESCHEDULE:
123         return new RescheduleTransferTask(argRule, delay, session.getRunner()
124                                                                  .getFileInformation(),
125                                           session);
126       case EXECJAVA:
127         return new ExecJavaTask(argRule, delay,
128                                 session.getRunner().getFileInformation(),
129                                 session);
130       case TRANSCODE:
131         return new TranscodeTask(argRule, delay,
132                                  session.getRunner().getFileInformation(),
133                                  session);
134       case SNMP:
135         return new SnmpTask(argRule, delay,
136                             session.getRunner().getFileInformation(), session);
137       case FTP:
138         return new FtpTransferTask(argRule, delay,
139                                    session.getRunner().getFileInformation(),
140                                    session);
141       case RENAME:
142         return new RenameTask(argRule, delay,
143                               session.getRunner().getFileInformation(),
144                               session);
145       case RESTART:
146         return new RestartServerTask(argRule, delay,
147                                      session.getRunner().getFileInformation(),
148                                      session);
149       case UNZEROED:
150         return new UnzeroedFileTask(argRule, delay,
151                                     session.getRunner().getFileInformation(),
152                                     session);
153       case CHMOD:
154         return new ChModTask(argRule, delay,
155                              session.getRunner().getFileInformation(), session);
156       case CHKFILE:
157         return new FileCheckTask(argRule, delay,
158                                  session.getRunner().getFileInformation(),
159                                  session);
160       case ICAP:
161         return new IcapTask(argRule, delay,
162                             session.getRunner().getFileInformation(), session);
163       default:
164         logger.error(NAME_UNKNOWN + type.name);
165         throw new OpenR66RunnerErrorException(UNVALID_TASK + type.name);
166     }
167   }

169   /**
170    * @param name
171    * @param argRule
172    * @param delay
173    * @param session
174    *
175    * @return the corresponding AbstractTask
176    *
177    * @throws OpenR66RunnerErrorException
178    */
179   public static AbstractTask getTaskFromId(final String name,
180                                            final String argRule,
181                                            final int delay,
182                                            final R66Session session)
183       throws OpenR66RunnerErrorException {
184     final TaskType type;
185     try {
186       type = valueOf(name);
187     } catch (final NullPointerException e) {
188       logger.error("name empty " + name);
189       throw new OpenR66RunnerErrorException(UNVALID_TASK + name);
190     } catch (final IllegalArgumentException e) {
191       logger.error(NAME_UNKNOWN + name);
192       throw new OpenR66RunnerErrorException(UNVALID_TASK + name);
193     }
194     return getTaskFromId(type, argRule, delay, session);
195   }

197   /**
198    * For usage in ExecBusinessTask
199    *
200    * @param name
201    * @param argRule
202    * @param delay
203    * @param session
204    *
205    * @return the corresponding AbstractTask
206    *
207    * @throws OpenR66RunnerErrorException
208    */
209   public static AbstractTask getTaskFromIdForBusiness(final String name,
210                                                       final String argRule,
211                                                       final int delay,
212                                                       final R66Session session)
213       throws OpenR66RunnerErrorException {
214     final TaskType type;
215     try {
216       type = valueOf(name);
217     } catch (final NullPointerException e) {
218       logger.error("name empty " + name);
219       throw new OpenR66RunnerErrorException(UNVALID_TASK + name);
220     } catch (final IllegalArgumentException e) {
221       logger.error(NAME_UNKNOWN + name);
222       throw new OpenR66RunnerErrorException(UNVALID_TASK + name);
223     }
224     switch (type) {
225       case LOG:
226         int newdelay = delay;
227         if (newdelay == 0) {
228           newdelay = 1;
229         }
230         return new LogTask(argRule, newdelay, "", session);
231       case EXEC:
232         return new ExecTask(argRule, delay, "", session);
233       case TRANSFER:
234         return new TransferTask(argRule, delay, "", session);
235       case TAR:
236         return new TarTask(argRule, delay, "", session);
237       case ZIP:
238         return new ZipTask(argRule, delay, "", session);
239       case EXECOUTPUT:
240         return new ExecOutputTask(argRule, delay, "", session);
241       case EXECJAVA:
242         return new ExecJavaTask(argRule, delay, "", session);
243       case SNMP:
244         return new SnmpTask(argRule, delay, "", session);
245       case FTP:
246         return new FtpTransferTask(argRule, delay, "", session);
247       case RESTART:
248         return new RestartServerTask(argRule, delay, "", session);
249       case ICAP:
250         return new IcapTask(argRule, delay, "", session);
251       case MOVE:
252       case MOVERENAME:
253       case COPY:
254       case COPYRENAME:
255       case EXECMOVE:
256       case LINKRENAME:
257       case VALIDFILEPATH:
258       case DELETE:
259       case RESCHEDULE:
260       case TRANSCODE:
261       case RENAME:
262       case UNZEROED:
263       case CHMOD:
264       case CHKFILE:
265         throw new OpenR66RunnerErrorException(UNVALID_TASK + type.name);
266       default:
267         logger.error(NAME_UNKNOWN + type.name);
268         throw new OpenR66RunnerErrorException(UNVALID_TASK + type.name);
269     }
270   }
