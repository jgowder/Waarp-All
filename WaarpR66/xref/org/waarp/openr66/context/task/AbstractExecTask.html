92    protected String applyTransferSubstitutions(String line) {
93      final Object[] argFormat = COMPILE_BLANK.split(argTransfer);
94      if (argFormat != null && argFormat.length > 0) {
95        try {
96          return String.format(line, argFormat);
97        } catch (final Exception e) {
98          // ignored error since bad argument in static rule info
99          logger.error("Bad format in Rule: {" + line + "} " + e.getMessage());
100       }
101     }
102     return line;
103   }

105   /**
106    * Generates a Command line object from rule and transfer data
107    *
108    * @param line the command to process as a string
109    */
110   protected CommandLine buildCommandLine(String line) {
111     if (line.contains(NOWAIT)) {
112       waitForValidation = false;
113     }
114     if (line.contains(LOCALEXEC)) {
115       useLocalExec = true;
116     }
117 
118     final String replacedLine =
119         COMPILE_REPLACE_ALL.matcher(line).replaceAll("\\${$1}");
120 
121     final CommandLine commandLine =
122         CommandLine.parse(replacedLine, getSubstitutionMap());
123 
124     final File exec = new File(commandLine.getExecutable());
125     if (exec.isAbsolute() && !exec.canExecute()) {
126       logger.error("Exec command is not executable: " + line);
127       final R66Result result =
128           new R66Result(session, false, ErrorCode.CommandNotFound,
129                         session.getRunner());
130       futureCompletion.setResult(result);
131       futureCompletion.cancel();
132       return null;
133     }
134 
135     return commandLine;
136   }

416       this.noOutput = noOutput;
417       this.waitForValidation = waitForValidation;
418     }

425       return commandLine;

429       return defaultExecutor;

433       return inputStream;

441       return pumpStreamHandler;

443 
444     public ExecuteWatchdog getWatchdog() {
445       return watchdog;

447 
448     public PrepareCommandExec invoke() {
449       commandLine = buildCommandLine(finalname);
450       if (commandLine == null) {
451         myResult = true;
452         return this;
453       }
454 
455       defaultExecutor = new DefaultExecutor();
456       if (noOutput) {
457         pumpStreamHandler = new PumpStreamHandler(null, null);
458       } else {
459         inputStream = new PipedInputStream();
460         outputStream = null;
461         try {
462           outputStream = new PipedOutputStream(inputStream);
463         } catch (final IOException e1) {
464           FileUtils.close(inputStream);
465           logger.error(
466               "Exception: " + e1.getMessage() + " Exec in error with " +
467               commandLine, e1);
468           futureCompletion.setFailure(e1);
469           myResult = true;
470           return this;
471         }
472         pumpStreamHandler = new PumpStreamHandler(outputStream, null);
473       }
474       defaultExecutor.setStreamHandler(pumpStreamHandler);
475       final int[] correctValues = { 0, 1 };
476       defaultExecutor.setExitValues(correctValues);
477       watchdog = null;
478       if (delay > 0 && waitForValidation) {
479         watchdog = new ExecuteWatchdog(delay);
480         defaultExecutor.setWatchdog(watchdog);
481       }
482       myResult = false;
483       return this;
484     }

583     private void closeAllForExecution(boolean interrupt) {
584       FileUtils.close(outputStream);
585       if (interrupt && thread != null) {
586         thread.interrupt();
587       }
588       FileUtils.close(inputStream);
589       try {
590         pumpStreamHandler.stop();
591       } catch (final IOException ignored) {
592         // nothing
593       }
594     }
