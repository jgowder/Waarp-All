21  package org.waarp.openr66.database.properties;
22  
23  import org.waarp.common.database.properties.DbProperties;
24  import org.waarp.common.database.properties.H2Properties;
25  import org.waarp.common.database.properties.MariaDBProperties;
26  import org.waarp.common.database.properties.MySQLProperties;
27  import org.waarp.common.database.properties.OracleProperties;
28  import org.waarp.common.database.properties.PostgreSQLProperties;
29  
30  public abstract class R66DbProperties {
31  
32    public String getCreateQuery() {
33      final StringBuilder builder = new StringBuilder();
34  
35      builder.append(
36          "CREATE TABLE MULTIPLEMONITOR (" + "COUNTCONFIG INTEGER NOT NULL, " +
37          "COUNTHOST INTEGER NOT NULL, " + "COUNTRULE INTEGER NOT NULL, " +
38          "HOSTID VARCHAR(8096) PRIMARY KEY);\n");
39      builder.append(
40          "CREATE TABLE CONFIGURATION (" + "READGLOBALLIMIT BIGINT NOT NULL, " +
41          "WRITEGLOBALLIMIT BIGINT NOT NULL," +
42          "READSESSIONLIMIT BIGINT NOT NULL," +
43          "WRITESESSIONLIMIT BIGINT NOT NULL," + "DELAYLIMIT BIGINT NOT NULL," +
44          "UPDATEDINFO INTEGER NOT NULL," + "HOSTID VARCHAR(8096) PRIMARY KEY);");
45      builder.append("CREATE TABLE HOSTCONFIG (" + "BUSINESS TEXT NOT NULL ," +
46                     "ROLES TEXT NOT NULL ," + "ALIASES TEXT NOT NULL ," +
47                     "OTHERS TEXT NOT NULL ," + "UPDATEDINFO INTEGER NOT NULL ," +
48                     "HOSTID VARCHAR(8096) PRIMARY KEY);\n");
49      builder.append("CREATE TABLE HOSTS (" + "ADDRESS VARCHAR(8096) NOT NULL ," +
50                     "PORT INTEGER NOT NULL ," + "ISSSL BOOLEAN NOT NULL ," +
51                     "HOSTKEY BYTEA NOT NULL ," + "ADMINROLE BOOLEAN NOT NULL ," +
52                     "ISCLIENT BOOLEAN NOT NULL ," +
53                     "ISACTIVE BOOLEAN NOT NULL ," +
54                     "ISPROXIFIED BOOLEAN NOT NULL ," +
55                     "UPDATEDINFO INTEGER NOT NULL ," +
56                     "HOSTID VARCHAR(8096) PRIMARY KEY);\n");
57      builder.append(
58          "CREATE TABLE RULES (" + "HOSTIDS TEXT ," + "MODETRANS INTEGER ," +
59          "RECVPATH VARCHAR(8096)," + "SENDPATH VARCHAR(8096) ," +
60          "ARCHIVEPATH VARCHAR(8096) ," + "WORKPATH VARCHAR(8096) ," +
61          "RPRETASKS TEXT ," + "RPOSTTASKS TEXT ," + "RERRORTASKS TEXT ," +
62          "SPRETASKS TEXT ," + "SPOSTTASKS TEXT ," + "SERRORTASKS TEXT ," +
63          "UPDATEDINFO INTEGER ," + "IDRULE VARCHAR(8096) PRIMARY KEY);\n");
64      builder.append("CREATE TABLE RUNNER (" + "GLOBALSTEP INTEGER NOT NULL ," +
65                     "GLOBALLASTSTEP INTEGER NOT NULL ," +
66                     "STEP INTEGER NOT NULL ," + "RANK INTEGER NOT NULL ," +
67                     "STEPSTATUS CHAR(3) NOT NULL ," +
68                     "RETRIEVEMODE BOOLEAN NOT NULL ," +
69                     "FILENAME VARCHAR(8096) NOT NULL ," +
70                     "ISMOVED BOOLEAN NOT NULL ," +
71                     "IDRULE VARCHAR(8096) NOT NULL ," +
72                     "BLOCKSZ INTEGER NOT NULL ," +
73                     "ORIGINALNAME VARCHAR(8096) NOT NULL ," +
74                     "FILEINFO TEXT NOT NULL ," + "TRANSFERINFO TEXT NOT NULL ," +
75                     "MODETRANS INTEGER NOT NULL ," +
76                     "STARTTRANS TIMESTAMP NOT NULL ," +
77                     "STOPTRANS TIMESTAMP NOT NULL ," +
78                     "INFOSTATUS CHAR(3) NOT NULL ," +
79                     "UPDATEDINFO INTEGER NOT NULL ," +
80                     "OWNERREQ VARCHAR(8096) NOT NULL ," +
81                     "REQUESTER VARCHAR(8096) NOT NULL ," +
82                     "REQUESTED VARCHAR(8096) NOT NULL ," +
83                     "SPECIALID BIGINT NOT NULL ," +
84                     "CONSTRAINT runner_pk PRIMARY KEY (OWNERREQ, REQUESTER, REQUESTED, SPECIALID));\n");
85      builder.append("CREATE INDEX IDX_RUNNER ON RUNNER (" +
86                     "STARTTRANS, OWNERREQ, STEPSTATUS, UPDATEDINFO, GLOBALSTEP, " +
87                     "INFOSTATUS, SPECIALID);\n");
88      builder.append(
89          "CREATE SEQUENCE RUNSEQ " + "MINVALUE -9223372036854775807 " +
90          "START WITH -9223372036854775807;\n");
91      return builder.toString();
92  
93    }
94  
95    public static R66DbProperties getInstance(final DbProperties prop) {
96      if (prop instanceof H2Properties) {
97        return new R66H2Properties();
98      } else if (prop instanceof MariaDBProperties) {
99        return new R66MariaDBProperties();
100     } else if (prop instanceof MySQLProperties) {
101       return new R66MySQLProperties();
102     } else if (prop instanceof PostgreSQLProperties) {
103       return new R66PostgreSQLProperties();
104     } else if (prop instanceof OracleProperties) {
105       return new R66OracleProperties();
106     } else {
107       return null;
108     }
109   }
110 }
