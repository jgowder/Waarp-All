467   /**
468    *
469    */
470   public class R66ClientGuiActions extends SwingWorker<String, Integer> {
471     static final int CHECKCONNECTION = 1;
472     static final int STARTTRANSFER = 2;
473     static final int FILESELECT = 3;
474     final int method;
475 
476     R66ClientGuiActions(final int method) {
477       this.method = method;
478       if (logger == null) {
479         logger = WaarpLoggerFactory.getLogger(R66ClientGui.class);
480       }
481     }
482 
483     @Override
484     protected String doInBackground() throws Exception {
485       disableAllButtons();
486       startRequest();
487       switch (method) {
488         case CHECKCONNECTION:
489           checkConnection();
490           break;
491         case STARTTRANSFER:
492           startTransfer();
493           break;
494         case FILESELECT:
495           findFile();
496           break;
497         default:
498           environnement.guiResultat =
499               Messages.getString("R66ClientGui.24"); //$NON-NLS-1$
500       }
501       setStatus(environnement.guiResultat);
502       if (method != FILESELECT) {
503         showDialog();
504       } else {
505         enableAllButtons();
506       }
507       stopRequest();
508       return environnement.guiResultat;
509     }
510   }
511 
512   private void showDialog() {
513     disableAllButtons();
514     if (dialog != null) {
515       dialog.dispose();
516       dialog = null;
517     }
518     dialog = new R66Dialog();
519     dialog.setLocationRelativeTo(frmRClientGui);
520     if (dialog.isAlwaysOnTopSupported()) {
521       dialog.setAlwaysOnTop(true);
522     } else {
523       dialog.toFront();
524     }
525     dialog.textPaneDialog.setText(environnement.guiResultat);
526     dialog.setVisible(true);
527     dialog.requestFocus();
528   }
529 
530   private void setStatus(final String mesg) {
531     textFieldStatus.setText(mesg);
532   }
533 
534   private void startRequest() {
535     progressBarTransfer.setIndeterminate(true);
536     progressBarTransfer.setValue(0);
537     progressBarTransfer.setVisible(true);
538     textPaneLog.setText("");
539   }
540 
541   private void stopRequest() {
542     progressBarTransfer.setIndeterminate(true);
543     progressBarTransfer.setValue(0);
544     progressBarTransfer.setVisible(false);
545     frmRClientGui.toFront();
546     frmRClientGui.requestFocus();
547   }

549   private void checkConnection() {
550     startRequest();
551     disableAllButtons();
552     environnement.hostId = (String) comboBoxHosts.getSelectedItem();
553     if (environnement.hostId == null || environnement.hostId.trim().isEmpty()) {
554       environnement.hostId =
555           Messages.getString("R66ClientGui.26"); //$NON-NLS-1$
556       environnement.guiResultat =
557           Messages.getString("R66ClientGui.27"); //$NON-NLS-1$
558     } else {
559       environnement.hostId = environnement.hostId.trim();
560       environnement.checkConnection();
561     }
562     setStatus(environnement.guiResultat);
563     showDialog();
564     stopRequest();
565   }
566 
567   private void startTransfer() {
568     logger.debug("start startTransfer");
569     disableAllButtons();
570     environnement.hostId = (String) comboBoxHosts.getSelectedItem();
571     environnement.ruleId = (String) comboBoxRules.getSelectedItem();
572     environnement.filePath = textFieldFile.getText();
573     environnement.information = textFieldInformation.getText();
574     environnement.isMD5 = checkBoxMD5.isSelected();
575 
576     boolean ok = true;
577     if (environnement.hostId == null || environnement.hostId.trim().isEmpty()) {
578       environnement.hostId =
579           Messages.getString("R66ClientGui.26"); //$NON-NLS-1$
580       ok = false;
581     } else {
582       environnement.hostId = environnement.hostId.trim();
583     }
584     if (environnement.filePath == null ||
585         environnement.filePath.trim().isEmpty()) {
586       environnement.filePath =
587           Messages.getString("R66ClientGui.30"); //$NON-NLS-1$
588       ok = false;
589     } else {
590       environnement.filePath = environnement.filePath.trim();
591     }
592     if (environnement.information == null ||
593         environnement.information.trim().isEmpty()) {
594       environnement.information = "";
595     } else {
596       environnement.information = environnement.information.trim();
597     }
598     if (environnement.ruleId == null || environnement.ruleId.trim().isEmpty()) {
599       environnement.ruleId =
600           Messages.getString("R66ClientGui.32"); //$NON-NLS-1$
601       ok = false;
602     } else {
603       environnement.ruleId = environnement.ruleId.trim();
604     }
605     if (ok) {
606       logger.debug("start startTransfer: " + environnement);
607       environnement.startsTransfer(progressBarTransfer, textFieldStatus);
608     } else {
609       environnement.guiResultat =
610           Messages.getString("R66ClientGui.34") + //$NON-NLS-1$

621   private void findFile() {
622     startRequest();
623     disableAllButtons();
624     try {
625       final JFileChooser fc = new JFileChooser();
626       final int returnVal = fc.showOpenDialog(frmRClientGui);
627       if (returnVal == JFileChooser.APPROVE_OPTION) {
628         final File file = fc.getSelectedFile();
629         try {
630           textFieldFile.setText(file.getCanonicalPath());
631         } catch (final IOException ignored) {
632           // nothing
633         }
634         setFindFile();
635         environnement.guiResultat =
636             Messages.getString("R66ClientGui.38"); //$NON-NLS-1$
637       }
638     } finally {
639       enableAllButtons();
640       stopRequest();
641     }
642   }

644   private void setFindFile() {
645     String text;
646     try {
647       text = textFieldFile.getText();
648     } catch (final NullPointerException e1) {
649       text = null;
650     }
651     if (text != null) {
652       final File file = new File(text);
653       if (file.exists()) {
654         text = file.toURI().toString();
655       } else {
656         // nothing
657       }
658       textFieldFile.setText(text);
659     }
660   }
