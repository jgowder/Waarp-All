125       packet.setCode(code.code);
126       session.newState(ERROR);
127       ChannelUtils
128           .writeAbstractLocalPacket(localChannelReference, packet, true);
129     } else {
130       session.newState(ERROR);
131       final ErrorPacketcalhandler/packet/ErrorPacket.html#ErrorPacket">ErrorPacket error = new ErrorPacket(

133           (e1 != null? e1.getMessage() : "Unknown Error") + " for " + packet +
134           " since " + code.getMesg(), code.getCode(),

142   /**
143    * Receive a request of Transfer
144    *
145    * @param packet
146    *
147    * @throws OpenR66ProtocolPacketException
148    * @throws OpenR66ProtocolBusinessException
149    * @throws OpenR66ProtocolSystemException
150    * @throws OpenR66RunnerErrorException
151    */
152   public void request(RequestPacket packet)
153       throws OpenR66ProtocolPacketException, OpenR66RunnerErrorException,
154              OpenR66ProtocolSystemException, OpenR66ProtocolBusinessException {
155     session.setStatus(99);
156     if (!session.isAuthenticated()) {
157       session.setStatus(48);
158       throw new OpenR66ProtocolNotAuthenticatedException(
159           Messages.getString("LocalServerHandler.3")); //$NON-NLS-1$
160     }
161     if (packet.isToValidate()) {
162       session.newState(REQUESTR);
163     }
164     // XXX validLimit only on requested side
165     if (checkRequest(packet)) {
166       return;
167     }
168     final DbRule rule;
169     try {
170       rule = new DbRule(packet.getRulename());
171     } catch (final WaarpDatabaseException e) {
172       logger
173           .info("Rule is unknown: {} {}", packet.getRulename(), e.getMessage());
174       session.setStatus(49);
175       endInitRequestInError(ErrorCode.QueryRemotelyUnknown, null,
176                             new OpenR66ProtocolBusinessException(
177                                 Messages.getString("LocalServerHandler.9") +
178                                 //$NON-NLS-1$
179                                 packet.getRulename()), packet);
180       return;
181     }
182     packet = computeBlockSizeFromRequest(packet, rule);
183     final DbTaskRunner runner;
184     // requested
185     final boolean isRetrieve = DbTaskRunner.getSenderByRequestPacket(packet);
186     if (packet.getSpecialId() != ILLEGALVALUE) {
187       runner = getPreviousDbTaskRunnerFromRequest(packet, rule, isRetrieve);
188       if (runner == null) {
189         return;
190       }
191     } else {
192       // Very new request
193       // should not be the case (the requester should always set the id)
194       logger.error("NO TransferID specified: SHOULD NOT BE THE CASE");
195       try {
196         runner = new DbTaskRunner(session, rule, isRetrieve, packet);
197       } catch (final WaarpDatabaseException e) {
198         session.setStatus(37);
199         endInitRequestInError(ErrorCode.QueryRemotelyUnknown, null,
200                               new OpenR66DatabaseGlobalException(e), packet);
201         return;
202       }
203       packet.setSpecialId(runner.getSpecialId());
204     }
205     runner.setSender(isRetrieve);
206     logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(),
207                  runner);
208     // Check now if request is a valid one
209     if (packet.getCode() != ErrorCode.InitOk.code) {
210       createErrorFromRequestInitKo(packet, runner);
211       return;
212     }
213     // Receiver can specify a rank different from database
214     setRankAtStartupFromRequest(packet, runner);
215     runner.setBlocksize(packet.getBlocksize());
216     try {
217       runner.update();
218     } catch (final WaarpDatabaseException ignored) {
219       // Ignore
220     }
221     logger
222         .debug("Filesize: {}:{}", packet.getOriginalSize(), runner.isSender());
223     boolean shouldInformBack = false;
224     try {
225       session.setRunner(runner);
226       // Fix to ensure that recv request are not trying to access to not chroot files
227       session.startup(Configuration.configuration.isChrootChecked() &&
228                       packet.isToValidate() && runner.isSender());
229       if (isRetrieve && !runner.isSendThrough()) {
230         if (packet.getOriginalSize() != runner.getOriginalSize()) {
231           packet.setOriginalSize(runner.getOriginalSize());
232           shouldInformBack = true;
233           logger.debug("Filesize2: {}:{}", packet.getOriginalSize(),
234                        runner.isSender());
235         }
236       }
237     } catch (final OpenR66RunnerErrorException e) {
238       try {
239         runner.saveStatus();
240       } catch (final OpenR66RunnerErrorException e1) {
241         logger.error("Cannot save Status: " + runner, e1);
242       }
243       if (runner.getErrorInfo() == ErrorCode.InitOk ||
244           runner.getErrorInfo() == ErrorCode.PreProcessingOk ||
245           runner.getErrorInfo() == ErrorCode.TransferOk) {
246         runner.setErrorExecutionStatus(ErrorCode.ExternalOp);
247       }
248       logger.error("PreTask in error {}", e.getMessage(), e);
249       errorToSend("PreTask in error: " + e.getMessage(), runner.getErrorInfo(),
250                   38);
251       return;
252     }
253     setFileSizeFromRequest(packet, runner, shouldInformBack);
254     session.setReady(true);
255     Configuration.configuration.getLocalTransaction()
256                                .setFromId(runner, localChannelReference);
257 
258     // Set read/write limit
259     final long remoteLimit = packet.getLimit();
260     long localLimit = localChannelReference.getChannelLimit(runner.isSender());
261     // Take the minimum speed
262     if (localLimit <= 0) {
263       localLimit = remoteLimit;
264     } else if (remoteLimit > 0 && remoteLimit < localLimit) {
265       localLimit = remoteLimit;
266     }
267     localChannelReference.setChannelLimit(runner.isSender(), localLimit);
268     packet.setLimit(localLimit);
269 
270     session.initializeDigest();
271     // inform back
272     informBackFromRequest(packet, runner);
273     if (!isRetrieve) {
274       prepareGlobalDigests();
275     }
276     // if retrieve => START the retrieve operation except if in Send Through mode
277     sendDataFromRequest(runner);
278     session.setStatus(39);
279   }

281   private RequestPacketopenr66/protocol/localhandler/packet/RequestPacket.html#RequestPacket">RequestPacket computeBlockSizeFromRequest(RequestPacket packet,
282                                                     final DbRule rule)
283       throws OpenR66ProtocolNotAuthenticatedException {
284     int blocksize = packet.getBlocksize();
285     if (packet.isToValidate()) {
286       if (!rule.checkHostAllow(session.getAuth().getUser())) {
287         session.setStatus(30);
288         throw new OpenR66ProtocolNotAuthenticatedException(
289             Messages.getString("LocalServerHandler.10")); //$NON-NLS-1$
290       }
291       // Check if the blocksize is greater than local value
292       if (Configuration.configuration.getBlockSize() < blocksize) {
293         logger.warn("Blocksize is greater than allowed {} < {}",
294                     Configuration.configuration.getBlockSize(), blocksize);
295         blocksize = Configuration.configuration.getBlockSize();
296         final String sep = localChannelReference.getPartner().getSeperator();
297         packet = new RequestPacket(packet.getRulename(), packet.getMode(),
298                                    packet.getFilename(), blocksize,
299                                    packet.getRank(), packet.getSpecialId(),
300                                    packet.getTransferInformation(),
301                                    packet.getOriginalSize(), sep);
302       }
303     }
304     if (!RequestPacket.isCompatibleMode(rule.getMode(), packet.getMode())) {
305       // not compatible Rule and mode in request
306       throw new OpenR66ProtocolNotAuthenticatedException(
307           Messages.getString("LocalServerHandler.12") + rule.getMode() + " vs "
308           //$NON-NLS-1$
309           + packet.getMode());
310     }
311     session.setBlockSize(blocksize);
312     return packet;
313   }

315   private boolean checkRequest(final RequestPacket packet)
316       throws OpenR66ProtocolPacketException {
317     if (packet.isToValidate()) {
318       if (Configuration.configuration.isShutdown()) {
319         logger.warn(Messages.getString("LocalServerHandler.7") //$NON-NLS-1$
320                     + packet.getRulename() + " from " + session.getAuth());
321         session.setStatus(100);
322         endInitRequestInError(ErrorCode.ServerOverloaded, null,
323                               new OpenR66ProtocolNotYetConnectionException(
324                                   "All new Request blocked"), packet);
325         session.setStatus(100);
326         return true;
327       }
328       if (Configuration.configuration.getConstraintLimitHandler()
329                                      .checkConstraints()) {
330         requestCheckConstraintsTrue(packet);
331         return true;
332       }
333     } else if (packet.getCode() == ErrorCode.ServerOverloaded.code) {
334       // XXX unvalid limit on requested host received
335       logger.info("TaskRunner initialisation in error: {}",
336                   ErrorCode.ServerOverloaded.getMesg());
337       localChannelReference.invalidateRequest(
338           new R66Result(null, session, true, ErrorCode.ServerOverloaded, null));
339       session.setStatus(101);
340       ChannelCloseTimer.closeFutureTransaction(this);
341       return true;
342     }
343     return false;
344   }

346   private static void setRankAtStartupFromRequest(final RequestPacket packet,
347                                                   final DbTaskRunner runner) {
348     if (runner.isSender()) {
349       logger.debug("Rank was: {} -> {}", runner.getRank(), packet.getRank());
350       runner.setRankAtStartup(packet.getRank());
351     } else {
352       if (runner.getRank() > packet.getRank()) {
353         logger.debug("Recv Rank was: {} -> {}", runner.getRank(),
354                      packet.getRank());
355         // if receiver, change only if current rank is upper proposed rank
356         runner.setRankAtStartup(packet.getRank());
357       }
358       if (packet.getOriginalSize() > 0) {
359         runner.setOriginalSize(packet.getOriginalSize());
360       }
361     }
362   }

364   private void setFileSizeFromRequest(final RequestPacket packet,
365                                       final DbTaskRunner runner,
366                                       boolean shouldInformBack)
367       throws OpenR66ProtocolPacketException {
368     logger
369         .debug("Filesize: {}:{}", packet.getOriginalSize(), runner.isSender());
370     if (!shouldInformBack) {
371       shouldInformBack =
372           !packet.getTransferInformation().equals(runner.getFileInformation());
373     }
374     if (runner.isFileMoved() && runner.isSender() && runner.isInTransfer() &&
375         runner.getRank() == 0 && !packet.isToValidate()) {
376       // File was moved during PreTask and very beginning of the transfer
377       // and the remote host has already received the request packet
378       // => Informs the receiver of the new name
379       sendFilenameFilesizeChanging(packet, runner,
380                                    "Will send a modification of filename due to pretask: ",
381                                    "Change Filename by Pre action on sender");
382     } else if (!packet.getFilename().equals(runner.getOriginalFilename()) &&
383                runner.isSender() && runner.isInTransfer() &&
384                runner.getRank() == 0 && !packet.isToValidate()) {
385       // File was modify at the very beginning (using wildcards)
386       // and the remote host has already received the request packet
387       // => Informs the receiver of the new name
388       sendFilenameFilesizeChanging(packet, runner,
389                                    "Will send a modification of filename due to wildcard: ",
390                                    "Change Filename by Wildcard on sender");
391     } else if (runner.isSelfRequest() && runner.isSender() &&
392                runner.isInTransfer() && runner.getRank() == 0 &&
393                !packet.isToValidate()) {
394       // FIX SelfRequest
395       // File could be modified at the very beginning (using wildcards)
396       // and the remote host has already received the request packet
397       // => Informs the receiver of the new name
398       sendFilenameFilesizeChanging(packet, runner,
399                                    "Will send a modification of filename due to wildcard in SelfMode: ",
400                                    "Change Filename by Wildcard on sender in SelfMode");
401     } else if (shouldInformBack && !packet.isToValidate()) {
402       // Was only (shouldInformBack)
403       // File length is now known, so inform back
404       sendFilenameFilesizeChanging(packet, runner,
405                                    "Will send a modification of filesize or fileInfo: ",
406                                    "Change Filesize / FileInfo on sender");
407     }
408   }

410   private void informBackFromRequest(final RequestPacket packet,
411                                      final DbTaskRunner runner)
412       throws OpenR66ProtocolPacketException {
413     if (packet.isToValidate()) {
414       if (Configuration.configuration.getMonitoring() != null) {
415         Configuration.configuration.getMonitoring().lastInActiveTransfer =
416             System.currentTimeMillis();
417       }
418       if (runner.isSender()) {
419         // In case Wildcard was used
420         logger.debug("New FILENAME: {}", runner.getOriginalFilename());
421         packet.setFilename(runner.getOriginalFilename());
422         logger.debug("Rank set: {}", runner.getRank());
423         packet.setRank(runner.getRank());
424       } else {
425         logger.debug("Rank set: {}", runner.getRank());
426         packet.setRank(runner.getRank());
427       }
428       packet.validate();
429       session.newState(REQUESTD);
430       ChannelUtils
431           .writeAbstractLocalPacket(localChannelReference, packet, true);
432     } else {
433       session.newState(REQUESTD);
434       // requester => might be a client
435       // Save the runner into the session and validate the request so begin transfer
436       session.getLocalChannelReference().getFutureRequest().setRunner(runner);
437       localChannelReference.getFutureValidRequest().setSuccess();
438       if (Configuration.configuration.getMonitoring() != null) {
439         Configuration.configuration.getMonitoring().lastOutActiveTransfer =
440             System.currentTimeMillis();
441       }
442     }
443   }

445   private void sendDataFromRequest(final DbTaskRunner runner) {
446     if (runner.isSender()) {
447       if (runner.isSendThrough()) {
448         // it is legal to send data from now
449         logger.debug("Now ready to continue with send through");
450         localChannelReference.validateEndTransfer(
451             new R66Result(session, false, ErrorCode.PreProcessingOk, runner));
452       } else {
453         // Automatically send data now
454         logger.debug("Now ready to continue with runRetrieve");
455         NetworkTransaction.runRetrieve(session);
456       }
457     }

479   private DbTaskRunner getPreviousDbTaskRunnerFromRequest(
480       final RequestPacket packet, final DbRule rule, final boolean isRetrieve)
481       throws OpenR66ProtocolPacketException {
482     final DbTaskRunner runner;// Reload or create
483     final String requested = DbTaskRunner.getRequested(session, packet);
484     final String requester = DbTaskRunner.getRequester(session, packet);
485     logger.debug("DEBUG: {}:{}", packet.getSpecialId(), isRetrieve);
486     if (packet.isToValidate()) {
487       // Id could be a creation or a reload
488       // Try reload
489       runner =
490           reloadDbTaskRunner(packet, rule, isRetrieve, requested, requester);
491       if (runner == null) {
492         return null;
493       }
494       final LocalChannelReference lcr =
495           Configuration.configuration.getLocalTransaction().getFromRequest(
496               requested + ' ' + requester + ' ' + packet.getSpecialId());
497       if (checkRunnerConsistency(packet, runner, lcr)) {
498         return null;
499       }
500 
501       if (runner.isAllDone()) {
502         // truly an error since done
503         session.setStatus(31);
504         endInitRequestInError(ErrorCode.QueryAlreadyFinished, runner,
505                               new OpenR66ProtocolBusinessQueryAlreadyFinishedException(
506                                   Messages.getString("LocalServerHandler.13")
507                                   //$NON-NLS-1$
508                                   + packet.getSpecialId()), packet);
509         return null;
510       }
511       if (lcr != null) {
512         // truly an error since still running
513         session.setStatus(32);
514         endInitRequestInError(ErrorCode.QueryStillRunning, runner,
515                               new OpenR66ProtocolBusinessQueryStillRunningException(
516                                   Messages.getString("LocalServerHandler.14")
517                                   //$NON-NLS-1$
518                                   + packet.getSpecialId()), packet);
519         return null;
520       }
521       logger.debug("Runner before any action: {} {}", runner.shallIgnoreSave(),
522                    runner);
523       // ok to restart
524       try {
525         if (runner.restart(false)) {
526           runner.saveStatus();
527         }
528       } catch (final OpenR66RunnerErrorException ignored) {
529         // nothing
530       }
531       // Change the SpecialID! => could generate an error ?
532       if (packet.getSpecialId() == ILLEGALVALUE) {
533         packet.setSpecialId(runner.getSpecialId());
534       }
535     } else {
536       // Id should be a reload
537       runner = reloadDbTaskRunnerFromId(packet, rule, isRetrieve, requested,
538                                         requester);
539     }
540     return runner;
541   }

543   private boolean checkRunnerConsistency(final RequestPacket packet,
544                                          final DbTaskRunner runner,
545                                          final LocalChannelReference lcr) {
546     // Check correctness of packet received vs current LCR
547     if (runner == null) {
548       logger.info("Id is unknown: {}}", packet.getSpecialId());
549       return true;
550     }
551     if (lcr != null && localChannelReference != null &&
552         !runner.isSelfRequest() &&
553         (!lcr.getLocalId().equals(localChannelReference.getLocalId()) ||
554          !lcr.getRemoteId().equals(localChannelReference.getRemoteId()))) {
555       logger.warn("LocalChannelReference differs: {}\n\t {}\n\tWill while " +
556                   "runner is AllDone: {}", localChannelReference, lcr,
557                   runner.isAllDone());
558       logger.info("Id is unknown: {}}", packet.getSpecialId());
559       if (runner.isAllDone()) {
560         try {
561           lcr.getServerHandler().tryFinalizeRequest(
562               new R66Result(lcr.getSession(), false, ErrorCode.Internal,
563                             runner));
564         } catch (final OpenR66RunnerErrorException ignore) {
565           SysErrLogger.FAKE_LOGGER.ignoreLog(ignore);
566         } catch (final OpenR66ProtocolSystemException ignore) {
567           SysErrLogger.FAKE_LOGGER.ignoreLog(ignore);
568         }
569         lcr.close();
570         if (localChannelReference.getClientRunner() != null &&
571             localChannelReference.getClientRunner().getTaskRunner() != null &&
572             localChannelReference.getClientRunner().getTaskRunner()
573                                  .isAllDone()) {
574           localChannelReference.close();
575         }
576         return true;
577       }
578       return false;
579     }
580     return false;
581   }

583   private DbTaskRunner reloadDbTaskRunnerFromId(final RequestPacket packet,
584                                                 final DbRule rule,
585                                                 final boolean isRetrieve,
586                                                 final String requested,
587                                                 final String requester)
588       throws OpenR66ProtocolPacketException {
589     DbTaskRunner runner;
590     try {
591       runner = new DbTaskRunner(session, rule, packet.getSpecialId(), requester,
592                                 requested);
593     } catch (final WaarpDatabaseException e) {
594       if (localChannelReference.getDbSession() == null) {
595         // Special case of no database client
596         try {
597           runner = new DbTaskRunner(session, rule, isRetrieve, packet);
598           logger.debug("Runner before any action: {} {}",
599                        runner.shallIgnoreSave(), runner);
600         } catch (final WaarpDatabaseException e1) {
601           session.setStatus(35);
602           endInitRequestInError(ErrorCode.QueryRemotelyUnknown, null,
603                                 new OpenR66DatabaseGlobalException(e1), packet);
604           return null;
605         }
606       } else {
607         endInitRequestInError(ErrorCode.QueryRemotelyUnknown, null,
608                               new OpenR66DatabaseGlobalException(e), packet);
609         session.setStatus(36);
610         return null;
611       }
612     }

798   public void data(final DataPacket packet)
799       throws OpenR66ProtocolNotAuthenticatedException,
800              OpenR66ProtocolBusinessException, OpenR66ProtocolPacketException {
801     if (!session.isAuthenticated()) {
802       logger.info("Not authenticated while Data received");
803       packet.clear();
804       throw new OpenR66ProtocolNotAuthenticatedException(
805           "Not authenticated while Data received");
806     }
807     if (!session.isReady()) {
808       logger.info("No request prepared");
809       packet.clear();
810       throw new OpenR66ProtocolBusinessException("No request prepared");
811     }
812     if (session.getRunner().isSender()) {
813       logger.error("Not in receive MODE but receive a packet");
814       packet.clear();
815       throw new OpenR66ProtocolBusinessException(
816           "Not in receive MODE but receive a packet");
817     }
818     if (!session.getRunner().continueTransfer()) {
819       logger.debug("EndTransfer failed ? {}",
820                    localChannelReference.getFutureEndTransfer().isFailed());
821       if (localChannelReference.getFutureEndTransfer().isFailed()) {
822         // nothing to do since already done
823         session.setStatus(94);
824         packet.clear();
825         return;
826       }
827       errorToSend("Transfer in error due previously aborted transmission",
828                   ErrorCode.TransferError, 95);
829       packet.clear();
830       return;
831     }
832     if (packet.getPacketRank() != session.getRunner().getRank()) {
833       logger.info("Issue on rank: {}:{}", packet.getPacketRank(),
834                   session.getRunner().getRank());

836         // cannot continue
837         logger.error(Messages.getString("LocalServerHandler.15") +
838                      packet.getPacketRank() + " : " + //$NON-NLS-1$
839                      session.getRunner().getRank() + " from {}",

843             ErrorCode.TransferError, 96);
844         packet.clear();
845         return;
846       }
847       // Fix the rank if possible
848       if (packet.getPacketRank() < session.getRunner().getRank()) {
849         logger.info("Bad RANK: {} : {}", packet.getPacketRank(),
850                     session.getRunner().getRank());
851         session.getRunner().setRankAtStartup(packet.getPacketRank());
852         session.getRestart().restartMarker(
853             session.getRunner().getBlocksize() * session.getRunner().getRank());
854         try {

856         } catch (final CommandAbstractException e) {

858                        session.getRunner().getRank());

860                       packet.getPacketRank(), ErrorCode.TransferError, 96);
861           packet.clear();
862           return;
863         }
864       } else {
865         // really bad
866         logger.error("Bad RANK: " + packet.getPacketRank() + " : " +
867                      session.getRunner().getRank());
868         errorToSend(
869             "Bad Rank in transmission: " + packet.getPacketRank() + " > " +
870             session.getRunner().getRank(), ErrorCode.TransferError, 20);
871         packet.clear();
872         return;
873       }
874     }
875     // Check global size
876     final long originalSize = session.getRunner().getOriginalSize();
877     if (originalSize >= 0) {
878       if (session.getRunner().getBlocksize() *
879           (session.getRunner().getRank() - 1) > originalSize) {
880         // cannot continue
881         logger.error(Messages.getString("LocalServerHandler.16") +
882                      packet.getPacketRank() + " : " + //$NON-NLS-1$
883                      (originalSize / session.getRunner().getBlocksize() + 1) +
884                      " from {}", session.getRunner());

886                     ErrorCode.TransferError, 96);

888         return;
889       }
890     }

903         if (!packet
904             .isKeyValid(session.getDigestBlock(), globalDigest, localDigest)) {
905           // Wrong packet
906           logger.error(Messages.getString("LocalServerHandler.17"), packet,
907                        //$NON-NLS-1$
908                        localChannelReference.getPartner()
909                                             .getDigestAlgo().algoName);
910           errorToSend("Transfer in error due to bad Hash on data packet " +
911                       "during multiple Digests (" +
912                       localChannelReference.getPartner()
913                                            .getDigestAlgo().algoName + ')',
914                       ErrorCode.MD5Error, 21);
915           packet.clear();
916           return;
917         }
918         // Only Packet digest and maybe localDigest
919       } else if (!packet
920           .isKeyValid(session.getDigestBlock(), null, localDigest)) {
921         // Wrong packet
922         logger.error(Messages.getString("LocalServerHandler.17"), packet,
923                      //$NON-NLS-1$
924                      localChannelReference.getPartner()
925                                           .getDigestAlgo().algoName);
926         errorToSend("Transfer in error due to bad Hash on data packet (" +
927                     localChannelReference.getPartner()
928                                          .getDigestAlgo().algoName + ')',
929                     ErrorCode.MD5Error, 21);
930         packet.clear();
931         return;
932       }
933     } else if (Configuration.configuration.isGlobalDigest()) {
934       // Only Global digests
935       FileUtils.computeGlobalHash(globalDigest, localDigest, packet.getData());
936     }
937     if (session.getRunner().isRecvThrough() &&
938         localChannelReference.isRecvThroughMode()) {
939       try {
940         localChannelReference.getRecvThroughHandler()
941                              .writeBytes(packet.getData());
942         session.getRunner().incrementRank();
943         if (packet.getPacketRank() % 100 == 1) {
944           logger.debug("Good RANK: {} : {}", packet.getPacketRank(),
945                        session.getRunner().getRank());
946         }
947       } finally {
948         packet.clear();
949       }
950     } else {
951       final DataBlock.html#DataBlock">DataBlock dataBlock = new DataBlock();
952       dataBlock.setBlock(packet.getData());
953       try {
954         session.getFile().writeDataBlock(dataBlock);
955         session.getRunner().incrementRank();
956         if (packet.getPacketRank() % 100 == 1) {
957           logger.debug("Good RANK: {} : {}", packet.getPacketRank(),
958                        session.getRunner().getRank());
959         }
960       } catch (final FileTransferException e) {
961         errorToSend("Transfer in error", ErrorCode.TransferError, 22);
962       } finally {
963         dataBlock.clear();
964         packet.clear();
965       }
966     }

1270             session.getBusinessObject() != null) {

1272             session.getBusinessObject().setInfo(session, optional);
1273           } else {
1274             final String temp = session.getBusinessObject().getInfo(session);
1275             session.getBusinessObject().setInfo(session, optional);
1276             optional = temp;
1277           }
1278         } else if (session.getExtendedProtocol() &&
1279                    transfer.getResult().getOther() == null &&
1280                    optional != null) {
1281           transfer.getResult().setOther(optional);
1282         }
1283         localChannelReference.validateRequest(transfer.getResult());
1284       }
1285     }
1286     session.setStatus(1);
1287     if (packet.isToValidate()) {
1288       session.newState(ENDREQUESTS);
1289       packet.validate();
1290       if (session.getExtendedProtocol()) {
1291         packet.setOptional(optional);
1292       }
1293       session.newState(ENDREQUESTR);
1294       try {
1295         ChannelUtils
1296             .writeAbstractLocalPacket(localChannelReference, packet, true);
1297       } catch (final OpenR66ProtocolPacketException ignored) {
1298         // nothing
1299       }
1300     } else {
1301       if (session.getState() != CLOSEDCHANNEL) {
1302         session.newState(ENDREQUESTR);
1303       }
1304     }
1305     if (runner != null &&
1306         (runner.isSelfRequested() || runner.isSelfRequest())) {
1307       ChannelCloseTimer.closeFutureTransaction(this);
1308     }
1309   }
