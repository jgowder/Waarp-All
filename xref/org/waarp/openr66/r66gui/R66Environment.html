77    public void initialize(final String[] args) {
78      WaarpLoggerFactory
79          .setDefaultFactoryIfNotSame(new WaarpSlf4JLoggerFactory(null));
80      initLog();
81      if (args.length < 1) {
82        SysErrLogger.FAKE_LOGGER
83            .syserr(Messages.getString("Configuration.WrongInit")); //$NON
84        // -NLS-1$
85        System.exit(2);//NOSONAR
86      }
87      if (!FileBasedConfiguration
88          .setClientConfigurationFromXml(Configuration.configuration, args[0])) {
89        logger.error(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
90        if (admin != null) {
91          admin.close();
92        }
93        if (DetectionUtils.isJunit()) {
94          return;
95        }
96        ChannelUtils.stopLogger();
97        System.exit(2);//NOSONAR
98      }
99      Configuration.configuration.pipelineInit();
100     networkTransaction = new NetworkTransaction();
101   }

140   public boolean startsTransfer(final JProgressBar progressBar,
141                                 final JEditorPane textFieldStatus) {
142     logger.debug("start startTransfer2");
143     final long time1 = System.currentTimeMillis();
144     final R66Futures/R66Future.html#R66Future">R66Future future = new R66Future(true);
145     final ProgressDirectTransfer transaction =
146         new ProgressDirectTransfer(future, hostId, filePath, ruleId,
147                                    information, isMD5,
148                                    Configuration.configuration.getBlockSize(),
149                                    ILLEGALVALUE, networkTransaction, 500,
150                                    progressBar, textFieldStatus);
151     logger.info("Launch transfer: " + hostId + ':' + ruleId + ':' + filePath);
152     transaction.run();
153     future.awaitOrInterruptible();
154     progressBar.setIndeterminate(true);
155     progressBar.setValue(0);
156     progressBar.setVisible(false);
157     final long time2 = System.currentTimeMillis();
158     final long delay = time2 - time1;
159     final R66Result result = future.getResult();
160     if (future.isSuccess()) {
161       if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
162         guiResultat = Messages.getString("R66Environment.8") + //$NON-NLS-1$
163                       result.getRunner().toShortNoHtmlString("<br>") +
164                       Messages.getString("R66Environment.10") + //$NON-NLS-2$
165                       hostId + (result.getFile() != null?
166             String.valueOf(result.getFile()) :
167             Messages.getString("R66ClientGui.30"))
168                       //$NON-NLS-2$
169                       + Messages.getString("R66Environment.13") +
170                       //$NON-NLS-1$
171                       delay;
172       } else {
173         guiResultat = Messages.getString("R66Environment.14") + //$NON-NLS-1$
174                       result.getRunner().toShortNoHtmlString("<br>") +
175                       Messages.getString("R66Environment.10") + //$NON-NLS-2$
176                       hostId + (result.getFile() != null?
177             String.valueOf(result.getFile()) :
178             Messages.getString("R66ClientGui.30"))
179                       //$NON-NLS-2$
180                       + Messages.getString("R66Environment.13") +
181                       //$NON-NLS-1$
182                       delay;
183       }
184     } else {
185       if (result == null || result.getRunner() == null) {
186         guiResultat = Messages.getString("R66Environment.20") + //$NON-NLS-1$
187                       Messages.getString("R66Environment.10") + //$NON-NLS-1$
188                       hostId + "     " + future.getCause().getMessage();
189       } else if (result.getRunner().getErrorInfo() == ErrorCode.Warning) {
190         guiResultat = Messages.getString("R66Environment.23") + //$NON-NLS-1$
191                       result.getRunner().toShortNoHtmlString("<br>") +
192                       Messages.getString("R66Environment.10") + //$NON-NLS-2$
193                       hostId + "    " + future.getCause().getMessage();
194       } else {
195         guiResultat = Messages.getString("R66Environment.27") + //$NON-NLS-1$
196                       result.getRunner().toShortNoHtmlString("<br>") +
197                       Messages.getString("R66Environment.10") + //$NON-NLS-2$
198                       hostId + "    " + future.getCause().getMessage();
199       }
200     }
201     return future.isSuccess();

254   public static String[] getRules(final boolean sendMode) {
255     if (logger == null) {
256       logger = WaarpLoggerFactory.getLogger(R66ClientGui.class);
257     }
258     final String[] results;
259     final DbRule[] dbRules;
260     try {
261       dbRules = DbRule.getAllRules();
262     } catch (final WaarpDatabaseNoConnectionException e) {
263       results = new String[1];
264       results[0] = Messages.getString("R66Environment.34"); //$NON-NLS-1$
265       return results;
266     }
267     int len = 0;
268     for (final DbRule rule : dbRules) {
269       if (sendMode) {
270         if (rule.isSendMode()) {
271           len++;
272         }
273       } else {
274         if (rule.isRecvMode()) {
275           len++;
276         }
277       }
278     }
279     if (len == 0) {
280       results = new String[1];
281       results[0] = Messages.getString("R66Environment.34"); //$NON-NLS-1$
282       return results;
283     }
284     results = new String[len];
285     int i = 0;
286     for (final DbRule rule : dbRules) {
287       if (sendMode) {
288         if (rule.isSendMode()) {
289           results[i] = rule.getIdRule();
290           i++;
291         }
292       } else {
293         if (rule.isRecvMode()) {
294           results[i] = rule.getIdRule();
295           i++;
296         }
297       }
298     }
299     return results;
300   }
